const state = {
    testEmail: '',
    
    get isEmailValid() {
        const email = this.testEmail || '';
        return email.includes('@') && email.includes('.') && email.length > 5;
    }
};

    currentImage: 'https://example.com/image1.jpg',
    galleryImages: [
        'https://example.com/image1.jpg',
        'https://example.com/image2.jpg',
        'https://example.com/image3.jpg'

        return () => {
            if (url === '') {
                this.$set('currentImage', '');
            } else {

        };
    }
};

Dynamic Styling

Dynamic CSS styling with theme switching.

Style Binding
<div data-bind="style:backgroundColor:dynamicColor, style:color:textColor">
    Dynamic styling
</div>
const state = {
    dynamicColor: '#6c757d',
    textColor: '#ffffff',
    styleMessage: 'Dynamic styling with px64',
    
    setTheme(theme) {
        return () => {
            const themes = {
                muted: { bg: '#6c757d', text: '#ffffff', msg: 'Muted theme' },
                dark: { bg: '#343a40', text: '#ffffff', msg: 'Dark theme' },
                light: { bg: '#f8f9fa', text: '#212529', msg: 'Light theme' }
            };
            const t = themes[theme];
            if (t) {
                this.$set('dynamicColor', t.bg);
                this.$set('textColor', t.text);
                this.$set('styleMessage', t.msg);
            }
        };
    }
};

px64.bind('#app', state);

Progress Bars
<div data-bind="progress:uploadProgress" class="progress"></div>
const state = {
    uploadProgress: 0,
    
    startUpload() {
        this.$set('uploadProgress', 0);
        const interval = setInterval(() => {
            const current = this.uploadProgress + Math.random() * 15;
            if (current >= 100) {
                this.$set('uploadProgress', 100);
                clearInterval(interval);
                setTimeout(() => this.$set('uploadProgress', 0), 2000);
            } else {
                this.$set('uploadProgress', current);
            }
        }, 200);
    }
};

px64.bind('#app', state);

Data Tables

Advanced table rendering with sorting, paging, and search.

Employee Directory
Page of 
<table data-bind="table:employees" 
       data-meta="cols:name,email,department,status,lastActive;sort:name" 
       class="table table-hover">
</table>
const state = {
    employees: px64.listState([]),
    searchTerm: '',
    
    generateEmployees() {
        const departments = ['Engineering', 'Marketing', 'Sales', 'HR'];
        const statuses = ['active', 'inactive', 'pending', 'success'];
        const employees = [];
        
        for (let i = 1; i <= 50; i++) {
            employees.push({
                id: i,
                name: `Employee ${i}`,
                email: `employee${i}@company.com`,
                department: departments[i % departments.length],
                status: statuses[i % statuses.length],
                lastActive: new Date(Date.now() - Math.random() * 86400000 * 30)
            });
        }
        
        this.employees.setItems(employees);
    }
};

px64.bind('#app', state);
